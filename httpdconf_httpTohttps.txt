LoadModule ssl_module modules/mod_ssl.so
Listen 443

<VirtualHost *:80>
   Redirect permanent / https://{{SERVER_NAME}}:443
</VirtualHost>

<VirtualHost *:443>
  ServerName {{SERVER_NAME}}
  SSLEngine on
  SSLCertificateFile /etc/httpd/ssl/myserver.crt
  SSLCertificateKeyFile /etc/httpd/ssl/myserver.key
</VirtualHost>
